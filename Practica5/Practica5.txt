utils.c

#include "ports.h"

/* Función que realiza una espera activa
* de time milisegundos
*/

void Delay(int time)
{
	static int delayLoopCount=400;
	int i;

	// Bucle de espera activa
	for( ; time>0; time--)
		for( i=0; i < delayLoopCount;i++ );
}
